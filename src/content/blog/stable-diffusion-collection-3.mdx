import { Image, getImage } from 'astro:assets'

import { optimiseImages } from '@/lib/image'

import ImageGallery from '@/components/ImageGallery.vue'
import ImageGalleryButton from '@/components/ImageGalleryButton.vue'
import ImageGalleryWrapper from '@/components/ImageGalleryWrapper.astro'

import img0 from '@/assets/posts/sd3/0.jpg'
import img1 from '@/assets/posts/sd3/1.jpg'
import img2 from '@/assets/posts/sd3/2.jpg'
import img3 from '@/assets/posts/sd3/3.jpg'
import img4 from '@/assets/posts/sd3/4.jpg'
import img5 from '@/assets/posts/sd3/5.jpg'
import img6 from '@/assets/posts/sd3/6.jpg'
import img7 from '@/assets/posts/sd3/7.jpg'
import img8 from '@/assets/posts/sd3/8.jpg'

export const alt = `Stable Diffusion generated image of an ancient alien.`

export const optimisedImages = await optimiseImages([
	{ src: img0, alt },
	{ src: img1, alt },
	{ src: img2, alt },
	{ src: img3, alt },
	{ src: img4, alt },
	{ src: img5, alt },
	{ src: img6, alt },
	{ src: img7, alt },
	{ src: img8, alt }
])

<ImageGalleryWrapper>
	{optimisedImages.map((optimisedImage, index) => (
		<ImageGalleryButton index={index} client:load>
			<Image
				src={optimisedImage.rawOptions.src} // uses the original image path
				alt={optimisedImage.attributes.alt}
				loading={index === 0 ? 'eager' : 'lazy'}
				quality={100}
				densities={[1.5, 2, 3]}
				class="h-full w-full object-cover"
			/>
		</ImageGalleryButton>
	))}
</ImageGalleryWrapper>

<ImageGallery images={optimisedImages} client:load />
